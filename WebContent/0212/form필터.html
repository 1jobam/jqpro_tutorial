:enabled : 사용가능, 접근가능, 실행가능
	:disabled : 사용불가능, 접근불가능, 실행불가능
	
	$('form :enabled').css('border','2px solid lime');
	$('form :disabled').css('border', '2px solid blue');
	
	상태값 가져오기
	prop('상태속성이름') : true 또는 false를 리턴한단
	
	상태값 변경
	prop('상태속성이름', true) : 해당 상태를 설정
	prop('상태속성이름', false) : 해당 상태를 해제
	
	prop('disabled', true) : 사용불가능으로 변경
	prop('disabled', false) : 사용가능으로 변경 - enabled를 사용하지 않는다
